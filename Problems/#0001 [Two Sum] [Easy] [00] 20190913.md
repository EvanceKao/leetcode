public class Solution {
    public int[] TwoSum(int[] nums, int target) {
        var result = new int[2];
       for (var i = 0;i < nums.Length;i++)
       {
           var cNum = nums[i];
           for (var j = i + 1;j < nums.Length;j++)
           {
               var addNum = nums[j];
               if (target == cNum + addNum)
               {
                   return new int[]{i, j};
               }
           }
       }

        return result;
    }
}
